const setProductSearch = (state, action) => {
    let productSearch = {
        isFetching: true,
        searchItems: [],
    }
 
    return productSearch;
}
 
const reducer = (state = initialState, action) => {
    switch (action.type) {
        case GET_INITIAL_PRODUCT_LIST:
        return { ...state, productSearch: setProductSearch(state, action) };
      case PRODUCT_LIST_RESULT:
        console.log('PRODUCTLIST_RESULT reducer: ',[...action.result.searchItems]);
        return {
          ...state,
          productSearch: {
            isFetching: false,
            searchItems: [...action.result.searchItems],
          },
 
        };
      case PRODUCT_LIST_ERROR:
        return { ...state, error: action.error };
        default:
            return state;
    }
};
 
export default reducer;
